19 sn_map* sn_map_create(SN_UINT capacity, sn_op* keys[], sn_mda* values[]) {
20  sn_map* obj = (sn_map*)SN_MALLOC(sizeof(sn_map));
21  obj->capacity = capacity;
22  obj->count = 0;
23  obj->keys = (sn_op**)SN_MALLOC(capacity * (sizeof(sn_op*) + sizeof(sn_mda*)));
24  obj->values = (sn_mda**)&(obj->keys[capacity]);
25  if (keys && values) {
26  for (SN_UINT i = 0; i < capacity; ++i) {
27  sn_map_insert(obj, keys[i], values[i]);
28  }
29  }
30  return obj;
31 }

33 sn_map* sn_map_from(SN_UINT capacity, ... /* sn_op*, sn_mda*, ... */) {
34  sn_map* obj = sn_map_create(capacity, NULL, NULL);

39  sn_mda* value = va_arg(vl, sn_mda*);

46 void sn_map_extend(sn_map* self, SN_UINT offset) {
47  sn_op** new_keys = (sn_op**)SN_MALLOC((self->capacity + offset) * (sizeof(sn_op*) + sizeof(sn_mda*)));
48  sn_mda** new_values = (sn_mda**)&(new_keys[self->capacity + offset]);
49  for (SN_UINT i = 0; i < self->count; ++i) {
50  new_keys[i] = self->keys[i];
51  new_values[i] = self->values[i];
52  }
53  SN_FREE(self->keys);
54  self->capacity += offset;
55  self->keys = new_keys;
56  self->values = new_values;
57 }

67 void sn_map_insert(sn_map* self, sn_op* key, sn_mda* value) {
68  if (self->count < self->capacity) {

89 sn_mda** sn_map_get_all(sn_map* self, sn_op* key) {

110 sn_op* sn_op_create(sn_op_type type, sn_flow_fn* flow, sn_dflow_fn* dflow, SN_UINT x_count, sn_op* x[]) {
111  sn_op* obj = (sn_op*)SN_MALLOC(sizeof(sn_op) + x_count * sizeof(sn_op*));
112  obj->ref_count = 1;
113  obj->type = type;
114  obj->flow = flow;
115  obj->dflow = dflow;
116  obj->x_count = x_count;
117  if (x) {
118  for (SN_UINT i = 0; i < x_count; ++i) {
119  ++(x[i]->ref_count);
120  obj->x[i] = x[i];
121  }
122  }
123  return obj;
124 }

149 static sn_mda* op_flow_(sn_op* self, sn_map* feed) {
150  sn_mda* y = NULL;
151  if (self->type == CONSTANT) {
152  y = sn_mda_copy(*((sn_mda**)(self->x)));
153  }
154  else if (self->type == OPERATOR) {
155  sn_mda** x = SN_DYNAMIC_ARRAY(sn_mda*, self->x_count);
156  for (SN_UINT i = 0; i < self->x_count; ++i) {
157  x[i] = op_flow_(self->x[i], feed);
158  sn_map_insert(feed, self, x[i]);
159  }
160  y = self->flow(self, x);
161  SN_FREE(x);
162  }
163  else if (self->type == PLACEHOLDER) {
164  y = sn_mda_copy(sn_map_get(feed, self));
165  }
166  else {
167  SN_ASSERT(false);
168  }
169  return y;
170 }

183 static sn_map* op_dflow_(sn_op* self, sn_map* feed) {
184  sn_map* dy_dx_map = NULL;
185  if (self->type == CONSTANT) {
186  /* !!! Not implemented !!! */
187  sn_mda* dy_dm = NULL;
188  }
189  else if (self->type == OPERATOR) {
190  sn_map** dm_dx_maps = SN_DYNAMIC_ARRAY(sn_map*, self->x_count);
191  SN_UINT dy_dx_map_capacity = 0;
192  for (SN_UINT i = 0; i < self->x_count; ++i) {
193  dm_dx_maps[i] = op_dflow_(self->x[i], feed);
194  dy_dx_map_capacity += dm_dx_maps[i]->count;
195  }
196 
197  sn_mda** x = sn_map_get_all(feed, self);
198  sn_mda* dy_dm = self->dflow(self, x);
199 
200  dy_dx_map = sn_map_create(dy_dx_map_capacity, NULL, NULL);
201  for (SN_UINT i = 0; i < self->x_count; ++i) {
202  for (SN_UINT j = 0; j < dm_dx_maps[i]->count; ++j) {
203  sn_map_insert(dy_dx_map, dm_dx_maps[i]->keys[j], sn_mda_gmatmul(dy_dm, dm_dx_maps[i]->values[j], x[i]->rank));
204  }
205  }
206 
207  SN_FREE(x);
208  for (SN_UINT i = 0; i < self->x_count; ++i) {
209  sn_map_destroy(dm_dx_maps[i]);
210  }
211  SN_FREE(dm_dx_maps);
212  sn_mda_destroy(dy_dm);
213  }
214  else if (self->type == PLACEHOLDER) {
215  sn_mda* feed_mda = sn_map_get(feed, self);
216  sn_mda* dy_dx = sn_mda_diagonal_full(feed_mda->rank, feed_mda->shape, 1.0);
217  dy_dx_map = sn_map_create(1, &self, &dy_dx);
218  }
219  else {
220  SN_ASSERT(false);
221  }
222 
223  return dy_dx_map;
224 }
