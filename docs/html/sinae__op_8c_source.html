40 static sn_mda* element_wise_binary_operator_flow_(sn_op* self, const sn_mda* x[], SN_FLOAT f(SN_FLOAT, SN_FLOAT)) {
41  SN_ASSERT((x[0]->rank == x[1]->rank || x[0]->rank * x[1]->rank == 0)); // If the rank of x0 and x1 does not match.
42  sn_mda* y = (x[0]->rank == 0) ? sn_mda_create(x[1]->rank, x[1]->shape) : sn_mda_create(x[0]->rank, x[0]->shape);
43  SN_UINT size = (x[0]->rank == 0) ? sn_mda_size(x[1]) : sn_mda_size(x[0]);
44  if (x[0]->rank == x[1]->rank) { // If both are in same shape.
45 #ifndef SN_NDEBUG
46  for (SN_UINT i = 0; i < x[0]->rank; ++i) {
47  SN_ASSERT(x[0]->shape[i] == x[1]->shape[i]);
48  }
49 #endif
50  for (SN_UINT i = 0; i < size; ++i) {
51  y->ptr[i] = f(x[0]->ptr[i], x[1]->ptr[i]);
52  }
53  }
54  else if (x[0]->rank == 0) { // If x[0] is scalar.
55  for (SN_UINT i = 0; i < size; ++i) {
56  y->ptr[i] = f(x[0]->ptr[0], x[1]->ptr[i]);
57  }
58  }
59  else if (x[1]->rank == 0) { // If x[1] is scalar.
60  for (SN_UINT i = 0; i < size; ++i) {
61  y->ptr[i] = f(x[0]->ptr[i], x[1]->ptr[0]);
62  }
63  }
64  // (x[0]->rank == 0) ? sn_mda_destroy(x[0]) : sn_mda_destroy(x[1]);
65  return y;
66 }
